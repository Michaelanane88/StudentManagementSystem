import java.util.ArrayList;

public class StudentManager {
    private ArrayList<Student> students;

    public StudentManager() {
        this.students = new ArrayList<>();
    }

    public void addStudent(Student student) {
        if (student == null) throw new IllegalArgumentException("student cannot be null");
        if (findStudent(student.getStudentId()) != null)
            throw new IllegalArgumentException("Duplicate studentId: " + student.getStudentId());
        students.add(student);
    }

    public boolean removeStudent(String studentId) {
        Student s = findStudent(studentId);
        if (s == null) return false;
        return students.remove(s);
    }

    public Student findStudent(String studentId) {
        if (studentId == null) return null;
        String id = studentId.trim();
        for (Student s : students) {
            if (s.getStudentId().equals(id)) return s;
        }
        return null;
    }

    public ArrayList<Student> getStudentsByMajor(String major) {
        ArrayList<Student> result = new ArrayList<>();
        if (major == null) return result;
        String m = major.trim();
        for (Student s : students) {
            if (s.getMajor().equalsIgnoreCase(m)) result.add(s);
        }
        return result;
    }

    public ArrayList<Student> getStudentsByYear(int year) {
        ArrayList<Student> result = new ArrayList<>();
        for (Student s : students) {
            if (s.getYear() == year) result.add(s);
        }
        return result;
    }

    public ArrayList<Student> getHonorStudents(double minGpa) {
        ArrayList<Student> result = new ArrayList<>();
        for (Student s : students) {
            if (s.getGpa() >= minGpa) result.add(s);
        }
        return result;
    }

    public double getAverageGpa() {
        if (students.isEmpty()) return 0.0;
        double sum = 0.0;
        for (Student s : students) sum += s.getGpa();
        return sum / students.size();
    }

    public double getAverageGpaByMajor(String major) {
        ArrayList<Student> list = getStudentsByMajor(major);
        if (list.isEmpty()) return 0.0;
        double sum = 0.0;
        for (Student s : list) sum += s.getGpa();
        return sum / list.size();
    }

    public void printAllStudents() {
        if (students.isEmpty()) {
            System.out.println("(No students)");
            return;
        }

        System.out.printf("%-6s %-18s %-28s %-6s %-20s %-4s%n",
                "ID", "Name", "Email", "GPA", "Major", "Year");
        System.out.println("-------------------------------------------------------------------------------------------");
        for (Student s : students) {
            System.out.printf("%-6s %-18s %-28s %-6.2f %-20s %-4d%n",
                    s.getStudentId(),
                    s.getFullName(),
                    s.getEmail(),
                    s.getGpa(),
                    s.getMajor(),
                    s.getYear());
        }
    }

    public int getTotalStudents() {
        return students.size();
    }

    public ArrayList<String> getAllMajors() {
        ArrayList<String> majors = new ArrayList<>();
        for (Student s : students) {
            String m = s.getMajor();
            if (m != null && !m.trim().isEmpty() && !containsIgnoreCase(majors, m.trim())) {
                majors.add(m.trim());
            }
        }
        return majors;
    }

    private boolean containsIgnoreCase(ArrayList<String> list, String value) {
        for (String s : list) {
            if (s.equalsIgnoreCase(value)) return true;
        }
        return false;
    }

    public ArrayList<Student> getAllStudentsCopy() {
        return new ArrayList<>(students);
    }
}

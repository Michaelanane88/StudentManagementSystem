import java.util.ArrayList;

public class ArrayListUtils {

    // Simple Predicate interface (avoids needing java.util.function.Predicate)
    public interface Predicate<T> {
        boolean test(T item);
    }

    public static <T> void swap(ArrayList<T> list, int index1, int index2) {
        if (list == null) throw new IllegalArgumentException("list is null");
        if (index1 < 0 || index1 >= list.size() || index2 < 0 || index2 >= list.size())
            throw new IndexOutOfBoundsException("Invalid indices");
        if (index1 == index2) return;
        T temp = list.get(index1);
        list.set(index1, list.get(index2));
        list.set(index2, temp);
    }

    public static <T extends Comparable<T>> T findMax(ArrayList<T> list) {
        if (list == null) throw new IllegalArgumentException("list is null");
        if (list.isEmpty()) return null;
        T max = list.get(0);
        for (int i = 1; i < list.size(); i++) {
            T current = list.get(i);
            if (current != null && (max == null || current.compareTo(max) > 0)) {
                max = current;
            }
        }
        return max;
    }

    public static <T> ArrayList<T> filter(ArrayList<T> list, Predicate<T> condition) {
        if (list == null) throw new IllegalArgumentException("list is null");
        if (condition == null) throw new IllegalArgumentException("condition is null");
        ArrayList<T> result = new ArrayList<>();
        for (T item : list) {
            if (condition.test(item)) result.add(item);
        }
        return result;
    }

    public static <T> void reverse(ArrayList<T> list) {
        if (list == null) throw new IllegalArgumentException("list is null");
        int left = 0, right = list.size() - 1;
        while (left < right) {
            swap(list, left, right);
            left++;
            right--;
        }
    }

    public static <T> ArrayList<T> merge(ArrayList<T> list1, ArrayList<T> list2) {
        ArrayList<T> result = new ArrayList<>();
        if (list1 != null) result.addAll(list1);
        if (list2 != null) result.addAll(list2);
        return result;
    }

    // Bounded generics for numbers
    public static <T extends Number> double sum(ArrayList<T> numbers) {
        if (numbers == null) throw new IllegalArgumentException("numbers is null");
        double total = 0.0;
        for (T n : numbers) {
            if (n != null) total += n.doubleValue();
        }
        return total;
    }

    public static <T extends Number> double average(ArrayList<T> numbers) {
        if (numbers == null) throw new IllegalArgumentException("numbers is null");
        if (numbers.isEmpty()) return 0.0;
        return sum(numbers) / numbers.size();
    }

    public static <T extends Number & Comparable<T>> ArrayList<T> filterAbove(ArrayList<T> numbers, T threshold) {
        if (numbers == null) throw new IllegalArgumentException("numbers is null");
        if (threshold == null) throw new IllegalArgumentException("threshold is null");
        ArrayList<T> result = new ArrayList<>();
        for (T n : numbers) {
            if (n != null && n.compareTo(threshold) > 0) result.add(n);
        }
        return result;
    }

    // Wildcards
    public static double sumNumbers(ArrayList<? extends Number> numbers) {
        if (numbers == null) throw new IllegalArgumentException("numbers is null");
        double total = 0.0;
        for (Number n : numbers) {
            if (n != null) total += n.doubleValue();
        }
        return total;
    }

    public static void addNumbers(ArrayList<? super Integer> list) {
        if (list == null) throw new IllegalArgumentException("list is null");
        // can add Integer values safely
        list.add(1);
        list.add(2);
        list.add(3);
    }

    public static void printList(ArrayList<?> list) {
        if (list == null) {
            System.out.println("(null list)");
            return;
        }
        for (Object o : list) {
            System.out.println(o);
        }
    }
}

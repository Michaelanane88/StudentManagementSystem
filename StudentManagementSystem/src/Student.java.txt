import java.util.Objects;

public class Student implements Comparable<Student> {
    private String studentId;
    private String firstName;
    private String lastName;
    private String email;
    private double gpa; // 0.0 - 4.0
    private String major;
    private int year; // 1-4

    public Student(String studentId, String firstName, String lastName, String email,
                   double gpa, String major, int year) {
        setStudentId(studentId);
        setFirstName(firstName);
        setLastName(lastName);
        setEmail(email);
        setGpa(gpa);
        setMajor(major);
        setYear(year);
    }

    public String getStudentId() { return studentId; }
    public void setStudentId(String studentId) {
        if (studentId == null || studentId.trim().isEmpty())
            throw new IllegalArgumentException("studentId cannot be null/empty");
        this.studentId = studentId.trim();
    }

    public String getFirstName() { return firstName; }
    public void setFirstName(String firstName) {
        if (firstName == null) firstName = "";
        this.firstName = firstName.trim();
    }

    public String getLastName() { return lastName; }
    public void setLastName(String lastName) {
        if (lastName == null) lastName = "";
        this.lastName = lastName.trim();
    }

    public String getEmail() { return email; }
    public void setEmail(String email) {
        if (email == null) email = "";
        this.email = email.trim();
    }

    public double getGpa() { return gpa; }
    public void setGpa(double gpa) {
        if (gpa < 0.0 || gpa > 4.0)
            throw new IllegalArgumentException("GPA must be 0.0..4.0");
        this.gpa = gpa;
    }

    public String getMajor() { return major; }
    public void setMajor(String major) {
        if (major == null) major = "";
        this.major = major.trim();
    }

    public int getYear() { return year; }
    public void setYear(int year) {
        if (year < 1 || year > 4)
            throw new IllegalArgumentException("year must be 1..4");
        this.year = year;
    }

    public String getFullName() {
        return (firstName + " " + lastName).trim();
    }

    @Override
    public String toString() {
        return String.format("Student{id=%s, name=%s, email=%s, gpa=%.2f, major=%s, year=%d}",
                studentId, getFullName(), email, gpa, major, year);
    }

    @Override
    public boolean equals(Object obj) {
        if (this == obj) return true;
        if (!(obj instanceof Student)) return false;
        Student other = (Student) obj;
        return Objects.equals(this.studentId, other.studentId);
    }

    @Override
    public int hashCode() {
        return Objects.hash(studentId);
    }

    /**
     * Natural ordering: GPA ascending, then studentId.
     * So ArrayListUtils.findMax(students) returns highest GPA student.
     */
    @Override
    public int compareTo(Student other) {
        if (other == null) return 1;
        int cmp = Double.compare(this.gpa, other.gpa);
        if (cmp != 0) return cmp;
        return this.studentId.compareTo(other.studentId);
    }
}
